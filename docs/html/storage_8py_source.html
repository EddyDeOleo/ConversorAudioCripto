85 def agregar_conversion(self, registro: dict):
86 """
87 @brief Agrega un registro de conversión al archivo.
88
89 @param registro Dict con campos obligatorios:
90 - archivo_origen
91 - ruta
92 - texto
93 - texto_encriptado (bytes o str)
94 - codigo_maquina (bytes o str)
95 - duracion_segundos
96 - info_audio
97
98 @return dict Registro final guardado.
99 @exception Exception Si ocurre un error al escribir el archivo.
100 """
101 data = self._leer_todo()
102 conversiones = data.get("conversiones", [])
103 next_id = (conversiones[-1]['id'] + 1) if conversiones else 1
104
105 # Normalizar campos que pueden venir en bytes
106 registro_final = {
107 "id": next_id,
108 "archivo_origen": registro.get("archivo_origen"),
109 "ruta": registro.get("ruta"),
110 "texto": registro.get("texto"),
111 "texto_encriptado": (
112 registro.get("texto_encriptado").decode('utf-8')
113 if isinstance(registro.get("texto_encriptado"), (bytes, bytearray))
114 else registro.get("texto_encriptado")
115 ),
116 "codigo_maquina": (
117 registro.get("codigo_maquina").decode('utf-8')
118 if isinstance(registro.get("codigo_maquina"), (bytes, bytearray))
119 else registro.get("codigo_maquina")
120 ),
121 "duracion_segundos": registro.get("duracion_segundos"),
122 "info_audio": registro.get("info_audio"),
123 "fecha": datetime.utcnow().isoformat() + "Z"
124 }
125
126 conversiones.append(registro_final)
127 data["conversiones"] = conversiones
128
129 exito = self._escribir_todo(data)
130 if exito:
131 logger.info(f"Registro agregado con id {next_id}")
