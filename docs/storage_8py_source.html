137 def _leer_todo(self):
138 """
139 @brief Lee el archivo JSON completo.

164 def agregar_conversion(self, registro: dict):
165 """
166 @brief Agrega un registro de conversión al archivo.
167
168 @param registro Dict con campos obligatorios:
169 - archivo_origen
170 - ruta
171 - texto
172 - texto_encriptado (bytes o str)
173 - codigo_maquina (bytes o str)
174 - duracion_segundos
175 - info_audio
176
177 @return dict Registro final guardado.
178 @exception Exception Si ocurre un error al escribir el archivo.
179 """
180 data = self._leer_todo()
181 conversiones = data.get("conversiones", [])
182 next_id = (conversiones[-1]['id'] + 1) if conversiones else 1
183
184 # Normalizar campos que pueden venir en bytes (Base64/UTF-8)
185 texto_encriptado_norm = registro.get("texto_encriptado")
186 if isinstance(texto_encriptado_norm, (bytes, bytearray)):
187 texto_encriptado_norm = texto_encriptado_norm.decode('utf-8')
188
189 codigo_maquina_norm = registro.get("codigo_maquina")
190 if isinstance(codigo_maquina_norm, (bytes, bytearray)):
191 # Asumo que el código máquina es Base64 para ser JSON-safe
192 try:
193 codigo_maquina_norm = base64.b64encode(codigo_maquina_norm).decode('utf-8')
194 except Exception:
195 # Si no es Base64, simplemente decodificar como UTF-8
196 codigo_maquina_norm = codigo_maquina_norm.decode('utf-8', errors='ignore')
197
198
199 registro_final = {
200 "id": next_id,
201 "archivo_origen": registro.get("archivo_origen"),
202 "ruta": registro.get("ruta"),
203 "texto": registro.get("texto"),
204 "texto_encriptado": texto_encriptado_norm,
205 "codigo_maquina": codigo_maquina_norm,
206 "duracion_segundos": registro.get("duracion_segundos"),
207 "info_audio": registro.get("info_audio"),
208 "fecha": datetime.utcnow().isoformat() + "Z"
209 }
210
211 conversiones.append(registro_final)
212 data["conversiones"] = conversiones
213
214 exito = self._escribir_todo(data)
215 if exito:
216 logger.info(f"Registro agregado con id {next_id}")
217 return registro_final
218 else:
219 raise Exception("No se pudo guardar la conversión en JSON")
